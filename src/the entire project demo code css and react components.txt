import React from 'react'; // Import React
import { BrowserRouter as Router, Route, Switch } from 'react-router-dom'; // Import React Router
import Navbar from './components/Navbar'; // Import Navbar component
import HeroHeader from './components/HeroHeader'; // Import HeroHeader component
import FeaturesList from './components/FeaturesList'; // Import FeaturesList component
import CTASection from './components/CTASection'; // Import CTASection component

import AboutSection from './components/AboutSection'; // Import AboutSection component
import TeamSection from './components/TeamSection'; // Import TeamSection component
import BenefitsSection from './components/BenefitsSection'; // Import BenefitsSection component
import StatsSection from './components/StatsSection'; // Import StatsSection component
import TestimonialSection from './components/TestimonialSection'; // Import TestimonialSection component
import PortfolioListSection from './components/PortfolioListSection'; // Import PortfolioListSection component
import FeatureSection from './components/FeatureSection'; // Import FeatureSection component
import GallerySection from './components/GallerySection'; // Import GallerySection component
import ContactForm from './components/ContactForm'; // Import ContactForm component
import LiveChatSupport from './components/LiveChatSupport'; // Import LiveChatSupport component
import SecureLogin from './components/SecureLogin'; // Import SecureLogin component
import Dashboard from './components/Dashboard'; // Import Dashboard component
import SupportTickets from './components/SupportTickets'; // Import SupportTickets component
import BillingInformation from './components/BillingInformation'; // Import BillingInformation component
import EventsCalendar from './components/EventsCalendar'; // Import EventsCalendar component
import ResourceCenter from './components/ResourceCenter'; // Import ResourceCenter component
import FAQs from './components/FAQs'; // Import FAQs component
import NewsletterSignup from './components/NewsletterSignup'; // Import NewsletterSignup component
import TestimonialsAndReviews from './components/TestimonialsAndReviews'; // Import TestimonialsAndReviews component
import './App.css'; // Import CSS for App

const App = () => {
  return (
    <Router>
      <div className="app">
        <Navbar /> {/* Navbar component */}
        <Switch>
          <Route exact path="/" component={HeroHeader} /> {/* Route for Home Page */}
          <Route path="/about-us" component={AboutSection} /> {/* Route for About Us Page */}
          <Route path="/services" component={FeaturesList} /> {/* Route for Services Page */}
          <Route path="/portfolio" component={PortfolioListSection} /> {/* Route for Portfolio Page */}
          <Route path="/contact-us" component={ContactForm} /> {/* Route for Contact Us Page */}

          <Route path="/newsletter-signup" component={NewsletterSignup} /> {/* Route for Newsletter Signup Page */}
          <Route path="/testimonials-reviews" component={TestimonialsAndReviews} /> {/* Route for Testimonials and Reviews Page */}
